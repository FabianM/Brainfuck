83 void brainfuck_parser_parse_string_partial(const char *string,
84  struct BrainfuckParserContext *ctx, int *error)
85 {
86  char ch;
87  int error_holder;
88  int i;
89  unsigned int n;
90  struct BrainfuckInstruction *instruction;
91 
92  assert(ctx);
93 
94  error = error ? error : &error_holder;
95  *error = BRAINFUCK_EOK;
96  instruction = ctx->tail;
97 
98  /* Initialise the context if it hasn't been initialised yet */
99  if (!ctx->loop && brainfuck_parser_context_init(ctx) != BRAINFUCK_EOK)
100  goto error_nomem;
101 
102  while((ch = *string++)) {
103  switch(ch) {
104  case '+':
105  case '-':
106  i = 0;
107  string--;
108  while ((ch = *string++) == '+' || ch == '-')
109  i += (ch == '+' ? 1 : -1);
110  string--;
111  if (i == 0)
112  break;
113  instruction = brainfuck_instruction_alloc();
114  instruction->type = INC;
115  instruction->argument.delta = i;
116  break;
117  case '>':
118  case '<':
119  i = 0;
120  string--;
121  while ((ch = *string++) == '>' || ch == '<')
122  i += (ch == '>' ? 1 : -1);
123  string--;
124  if (i == 0)
125  break;
126  instruction = brainfuck_instruction_alloc();
127  instruction->type = MOV;
128  instruction->argument.delta = i;
129  break;
130  case ',':
131  n = 0;
132  string--;
133  while ((ch = *string++) == ',' && n++);
134  instruction = brainfuck_instruction_alloc();
135  instruction->type = IN;
136  instruction->argument.n = n;
137  break;
138  case '.':
139  n = 0;
140  string--;
141  while ((ch = *string++) == '.' && n++);
142  instruction = brainfuck_instruction_alloc();
143  instruction->type = OUT;
144  instruction->argument.n = n;
145  break;
146  case '[':
147  instruction = brainfuck_instruction_alloc();
148  instruction->type = JZ;
149  instruction->argument.jump = NULL;
150  ctx->loop[ctx->loop_index++] = instruction;
151  break;
152  case ']':
153  if (ctx->loop_index <= 0)
154  goto error_syntax;
155 
156  instruction = brainfuck_instruction_alloc();
157  instruction->type = JMP;
158  instruction->argument.jump = ctx->loop[--ctx->loop_index];
159  instruction->argument.jump->argument.jump = instruction;
160  break;
161  default:
162  continue;
163  }
164 
165  if (!ctx->head)
166  ctx->head = instruction;
167  else
168  ctx->tail->next = instruction;
169  ctx->tail = instruction;
170  }
171  return;
172 
173  error_nomem:
174  *error = BRAINFUCK_ENOMEM;
175  return;
176  error_syntax:
177  *error = BRAINFUCK_ESYNTAX;
178  return;
179 }

181 struct BrainfuckScript * brainfuck_parser_parse_string(const char *string,
182  int *error)
183 {
184  int error_holder;
185  struct BrainfuckParserContext ctx = {NULL, NULL, NULL, 0, 0};
186 
187  error = error ? error : &error_holder;
188  *error = BRAINFUCK_EOK;
189 
190  brainfuck_parser_parse_string_partial(string, &ctx, error);
191  if (*error != BRAINFUCK_EOK)
192  return NULL;
193 
194  if (!brainfuck_parser_validate(&ctx)) {
195  *error = BRAINFUCK_ESYNTAX;
196  return NULL;
197  }
198  free(ctx.loop);
199  return ctx.head;
200 }

202 struct BrainfuckScript * brainfuck_parser_parse_file(FILE *file, int *error)
203 {
204  int error_holder = BRAINFUCK_EOK;
205  char buffer[BRAINFUCK_BUFFER_SIZE] = {0};
206  struct BrainfuckParserContext ctx = {NULL, NULL, NULL, 0, 0};
207 
208  error = error ? error : &error_holder;
209  *error = BRAINFUCK_EOK;
210 
211  assert(file);
212  while (!feof(file)) {
213  memset(&buffer, 0, sizeof(buffer));
214  fread(&buffer, sizeof(buffer), 1, file);
215  if (ferror(file))
216  continue;
217  brainfuck_parser_parse_string_partial(buffer, &ctx, error);
218  if (*error != BRAINFUCK_EOK)
219  return NULL;
220  }
221 
222  if (!brainfuck_parser_validate(&ctx)) {
223  *error = BRAINFUCK_ESYNTAX;
224  return NULL;
225  }
226  free(ctx.loop);
227  return ctx.head;
228 }

240 int brainfuck_engine_run(const struct BrainfuckScript *script,
241  struct BrainfuckEngineContext *ctx)
242 {
243  assert(script);
244  assert(ctx);
245 
246  unsigned int k;
247 
248  while (script) {
249  switch(script->type) {
250  case INC:
251  *ctx->memory += script->argument.delta;
252  break;
253  case MOV:
254  ctx->memory += script->argument.delta;
255  break;
256  case CLR:
257  *ctx->memory = 0;
258  break;
259  case IN:
260  for (k = 0; k < script->argument.n; k++)
261  *ctx->memory = ctx->read();
262  break;
263  case OUT:
264  for (k = 0; k < script->argument.n; k++)
265  ctx->write(*ctx->memory);
266  break;
267  case JMP:
268  script = script->argument.jump;
269  continue;
270  case JZ:
271  if (!*ctx->memory)
272  script = script->argument.jump;
273  break;
274  case NOP:
275  break;
276  }
277  script = script->next;
278  }
279  return BRAINFUCK_EOK;
280 }
