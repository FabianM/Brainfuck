34 /**
35  * Main entry point of the program.
36  */
37 int main()
38 {
39  /* Declaration */
40  struct BrainfuckScript *script;
41  struct BrainfuckEngineContext ctx;
42  int error;
43  unsigned int *memory;
44  FILE *file;
45 
46  /* Initialization */
47  ctx.read = &getchar;
48  ctx.write = &putchar;
49  ctx.memory = memory = calloc(30000, sizeof(unsigned int));
50  file = fopen("../examples/hanoi.bf", "r");
51 
52  /* Parsing */
53  script = brainfuck_parser_parse_file(file, &error);
54 
55  if (error != BRAINFUCK_EOK) {
56  fprintf(stderr, "error: parser exited with code %i.\n", error);
57  free(ctx.memory);
58  brainfuck_script_dealloc(script);
59  return 1;
60  }
61 
62  /* Execution */
63  error = brainfuck_engine_run(script, &ctx);
64 
65  /* Cleaning */
66  fclose(file);
67  free(memory);
68  brainfuck_script_dealloc(script);
69 
70  /* Exit */
71  return error != BRAINFUCK_EOK;
72 }
